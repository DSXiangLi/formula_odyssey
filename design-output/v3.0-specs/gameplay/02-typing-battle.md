## 6. 输入系统

### 6.1 实时匹配

```typescript
interface InputSystem {
  currentInput: string;           // 当前输入
  targetText: string;             // 目标文本

  // 处理键盘输入
  handleInput(char: string): void {
    this.currentInput += char;
    this.checkMatch();
  }

  // 检查匹配
  checkMatch(): void {
    if (this.currentInput === this.targetText) {
      // 完全匹配
      this.fireProjectile();
      this.clearInput();
    } else if (!this.targetText.startsWith(this.currentInput)) {
      // 输入错误
      this.showError();
    }
  }
}
```

```typescript
  convertPinyin(pinyin: string): string[] {
    // 使用pinyin库转换

  // 检查输入是否匹配目标
  checkMatch(input: string, target: string): boolean {
    // 直接匹配
```

```typescript
// 60fps目标
class BattleRenderer {
  private lastFrameTime: number = 0;

  gameLoop(currentTime: number): void {
    const deltaTime = (currentTime - this.lastFrameTime) / 1000;
    this.lastFrameTime = currentTime;

    // 更新敌人位置
    this.updateEnemies(deltaTime);

    // 检查碰撞
    this.checkCollisions();

    // 渲染
    this.render();

    // 下一帧
    requestAnimationFrame((time) => this.gameLoop(time));
  }
}
```

```typescript
// 防抖处理
class InputHandler {
  private keyBuffer: string[] = [];
  private lastProcessTime: number = 0;

  onKeyDown(event: KeyboardEvent): void {
    // 忽略特殊键
    if (this.isSpecialKey(event)) return;

    // 加入缓冲区
    this.keyBuffer.push(event.key);

    // 限制缓冲区大小
    if (this.keyBuffer.length > 10) {
      this.keyBuffer.shift();
    }
  }

  // 每帧处理输入
  processInput(): void {
    if (this.keyBuffer.length === 0) return;

    // 合并连续输入
    const input = this.keyBuffer.join('');
    this.keyBuffer = [];

    // 处理输入
    this.battleSystem.handleInput(input);
  }
}
```
